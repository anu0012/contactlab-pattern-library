Breadcrumb

Syntax


		<ol class="breadcrumb">
			<li><a href="#">item1</a></li>
			<li><a href="#">item2</a></li>
			<li class="active">item3</li>
		</ol>
	


		<ol class="breadcrumb">
			<li><a href="#"><i class="fa fa-home"></i> item1 </a></li>
			<li><a href="#"><i class="fa fa-book"></i> item2 </a></li>
			<li class="active"><i class="fa fa-cog"></i> item3 </li>
		</ol>
